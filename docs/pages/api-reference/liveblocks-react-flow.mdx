`@liveblocks/react-flow` provides you with [React Flow](https://reactflow.dev/)
bindings for our realtime collaboration APIs, built on top of WebSockets. Read
our [getting started](/docs/get-started) guides to learn more.

## Dependencies and styles

- Install peer dependency [`@xyflow/react`](https://reactflow.dev/) and include
  [React Flow’s base CSS](https://reactflow.dev/learn/troubleshooting/common-errors#styles-not-applied).
- Import `@liveblocks/react-flow/styles.css` when you use package UI such as
  [`Cursors`](#cursors).

## Suspense

Hooks from this package are available from two entry points, similar to
[`@liveblocks/react`](/docs/api-reference/liveblocks-react#Suspense):

- **`@liveblocks/react-flow`** — `useLiveblocksFlow` exposes `isLoading` and
  returns `nodes` and `edges` as `null` until Liveblocks Storage is ready.
- **`@liveblocks/react-flow/suspense`** — the export named `useLiveblocksFlow`
  suspends until Liveblocks Storage is ready. `nodes` and `edges` are always
  arrays; wrap the tree in `ClientSideSuspense` (or React `Suspense`) so loading
  is handled above.

```tsx highlight="1,4-5,13-17"
import { useLiveblocksFlow } from "@liveblocks/react-flow";

export function Flow() {
  const { nodes, edges, isLoading, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow();

  if (isLoading || nodes === null || edges === null) {
    return <div>Loading…</div>;
  }

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      onConnect={onConnect}
    />
  );
}
```

```tsx highlight="1-2,5-6,22-24"
import { ClientSideSuspense } from "@liveblocks/react/suspense";
import { useLiveblocksFlow } from "@liveblocks/react-flow/suspense";

function Flow() {
  const { nodes, edges, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow();

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      onConnect={onConnect}
    />
  );
}

export function App() {
  return (
    <RoomProvider id="my-room-id">
      <ClientSideSuspense fallback={<div>Loading…</div>}>
        <Flow />
      </ClientSideSuspense>
    </RoomProvider>
  );
}
```

## useLiveblocksFlow [@badge=RoomProvider]

Returns controlled [React Flow](https://reactflow.dev/) state backed by
Liveblocks Storage. Nodes and edges are stored in `LiveMap`s keyed by id so
multiple clients can edit the graph concurrently.

The hook initializes an empty flow (or your `nodes.initial` / `edges.initial`
defaults) the first time Storage loads for the room, in the same way as
[`initialStorage`](/docs/api-reference/liveblocks-react#RoomProvider) — it only
writes if the `storageKey` root is not already present.

```tsx highlight="4-5,10-27"
"use client";

import { ClientSideSuspense, RoomProvider } from "@liveblocks/react/suspense";
import { useLiveblocksFlow } from "@liveblocks/react-flow/suspense";
import { Controls, MiniMap, ReactFlow } from "@xyflow/react";

function Flow() {
  const { nodes, edges, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow({
      nodes: {
        initial: [
          {
            id: "1",
            type: "input",
            data: { label: "Node 1" },
            position: { x: 250, y: 25 },
          },
          {
            id: "2",
            data: { label: "Node 2" },
            position: { x: 100, y: 125 },
          },
        ],
      },
      edges: {
        initial: [{ id: "e1-2", source: "1", target: "2" }],
      },
    });

  return (
    <div className="h-screen w-screen">
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
        fitView
      >
        <MiniMap />
        <Controls />
      </ReactFlow>
    </div>
  );
}

export default function Page() {
  return (
    <RoomProvider id="my-room-id">
      <ClientSideSuspense fallback={null}>
        <Flow />
      </ClientSideSuspense>
    </RoomProvider>
  );
}
```

<PropertiesList title="Options">
  <PropertiesListItem name="nodes" type="{ initial?: Node[] }" optional>
    Default nodes used only when the room has no data at the given `storageKey`
    yet. Omit `initial` or pass an empty array for no default nodes.
  </PropertiesListItem>
  <PropertiesListItem name="edges" type="{ initial?: Edge[] }" optional>
    Default edges used only when the room has no data at the given `storageKey`
    yet. Omit `initial` or pass an empty array for no default edges.
  </PropertiesListItem>
  <PropertiesListItem name="storageKey" type="string" optional>
    Root key for this flow in Liveblocks Storage. Defaults to `"flow"`. See
    [`storageKey`](#storageKey).
  </PropertiesListItem>
</PropertiesList>

<Banner type="success" title="Options are not reactive">

The `nodes`, `edges`, `initial`, and `storageKey` values are read once when the
hook mounts. Later changes to those props do not reconfigure Storage.

</Banner>

<PropertiesList title="Returns">
  <PropertiesListItem name="nodes" type="Node[] | null">
    Current nodes merged from Storage and local-only fields. With
    `@liveblocks/react-flow`, this is `null` while Storage is loading; with
    `@liveblocks/react-flow/suspense`, it is always an array.
  </PropertiesListItem>
  <PropertiesListItem name="edges" type="Edge[] | null">
    Current edges merged from Storage and local-only fields. `null` only on the
    non-Suspense entry while loading.
  </PropertiesListItem>
  <PropertiesListItem name="isLoading" type="boolean">
    Whether Storage is still loading. Only meaningful when importing from
    `@liveblocks/react-flow` (always `false` when using the Suspense entry).
  </PropertiesListItem>
  <PropertiesListItem name="onNodesChange" type="OnNodesChange">
    Pass to React Flow’s `onNodesChange`. Persists structural and positional
    updates to Storage and keeps ephemeral UI state local.
  </PropertiesListItem>
  <PropertiesListItem name="onEdgesChange" type="OnEdgesChange">
    Pass to React Flow’s `onEdgesChange`.
  </PropertiesListItem>
  <PropertiesListItem name="onConnect" type="OnConnect">
    Pass to React Flow’s `onConnect`. Adds new edges to Storage (deduplicated by
    source, target, and handles).
  </PropertiesListItem>
</PropertiesList>

#### Local state vs Storage

Some React Flow fields are intentionally **not** written to Liveblocks Storage
so each client keeps their own selection and interaction state:

- **Nodes:** `selected`, `dragging`, `measured`, `resizing`
- **Edges:** `selected`

Everything else on nodes and edges (including `position`, stored width and
height, `data`, handles, and edge endpoints) is synchronized through Storage.

## Storage key [#storageKey]

By default, `useLiveblocksFlow` reads and writes the flow under the Storage root
key **`"flow"`**. Pass `storageKey` when you need a different property name—for
example **several independent graphs in the same room**, or to **avoid
colliding** with other features that use Storage on the same root object.

The value must match everywhere you seed or read that flow:

- The same string on **`useLiveblocksFlow({ storageKey: "…" })`**
- The matching key on **`initialStorage`** when using
  [`createLiveblocksFlow`](#createLiveblocksFlow)

```tsx highlight="6,21"
import { useLiveblocksFlow } from "@liveblocks/react-flow/suspense";
import { ReactFlow } from "@xyflow/react";

function FlowA() {
  const { nodes, edges, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow({ storageKey: "flowA" });

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      onConnect={onConnect}
    />
  );
}

function FlowB() {
  const { nodes, edges, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow({ storageKey: "flowB" });

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      onConnect={onConnect}
    />
  );
}
```

```tsx highlight="2,7-8"
import { RoomProvider } from "@liveblocks/react/suspense";
import { createLiveblocksFlow } from "@liveblocks/react-flow";

<RoomProvider
  id="my-room-id"
  initialStorage={{
    flowA: createLiveblocksFlow(nodesA, edgesA),
    flowB: createLiveblocksFlow(nodesB, edgesB),
  }}
>
  {/* Each hook uses the matching storageKey */}
</RoomProvider>;
```

In TypeScript, declare **both** keys (or a shared type) on
[`Liveblocks.Storage`](#TypeScript) so `useStorage` and autocomplete stay
accurate.

Like other hook options, `storageKey` is **fixed at mount**; changing the prop
later does not move data to another key.

## createLiveblocksFlow [#createLiveblocksFlow]

Builds the Liveblocks live structure tree for a flow so you can seed your Room
provider with `initialStorage` instead of (or in addition to) the hook’s
`nodes.initial` / `edges.initial` (or legacy `initial`) options. Use the
**same** property name as [`storageKey`](#storageKey) on
[`useLiveblocksFlow`](#useLiveblocksFlow) (default `flow`).

```tsx highlight="2,14"
import { RoomProvider } from "@liveblocks/react/suspense";
import { createLiveblocksFlow } from "@liveblocks/react-flow";

const initialNodes = [
  { id: "1", position: { x: 0, y: 0 }, data: { label: "Node 1" } },
];
const initialEdges = [{ id: "1-2", source: "1", target: "2" }];

export function App() {
  return (
    <RoomProvider
      id="my-room-id"
      initialStorage={{
        flow: createLiveblocksFlow(initialNodes, initialEdges),
      }}
    >
      {/* children */}
    </RoomProvider>
  );
}
```

## Custom nodes [#custom-nodes]

[Custom node types](https://reactflow.dev/learn/customization/custom-nodes) are
ordinary React Flow nodes: you pass a `nodeTypes` map to
[`ReactFlow`](https://reactflow.dev/api-reference/react-flow#nodetypes), and
each node in Storage sets `type` to the key for its component (for example
`type: "task"` maps to `nodeTypes={{ task: TaskNodeView }}`). Nothing
Liveblocks-specific is required beyond keeping stored fields valid for
[Storage](/docs/ready-made-features/multiplayer/sync-engine/liveblocks-storage)
(typically JSON-like `data` on the node).

For TypeScript, pass a **union** of your custom node types and a fallback such
as `Node<JsonObject>` as the first generic argument to `useLiveblocksFlow` so
`nodes` and handlers stay correctly typed:

```tsx highlight="11,34,37-42,54"
import type { JsonObject } from "@liveblocks/react";
import { useLiveblocksFlow } from "@liveblocks/react-flow/suspense";
import {
  type Node,
  type NodeProps,
  ReactFlow,
  useReactFlow,
} from "@xyflow/react";
import { memo, useCallback } from "react";

type TaskNode = Node<{ title: string }>;

const TaskNodeView = memo(({ id, data }: NodeProps<TaskNode>) => {
  const { updateNode } = useReactFlow();

  const rename = useCallback(() => {
    updateNode(id, (node) => ({
      ...node,
      data: { ...node.data, title: "Updated" },
    }));
  }, [id, updateNode]);

  return (
    <div>
      <button type="button" onClick={rename}>
        {data.title}
      </button>
    </div>
  );
});

function Flow() {
  const { nodes, edges, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow<TaskNode | Node<JsonObject>>({
      nodes: {
        initial: [
          {
            id: "1",
            type: "task",
            position: { x: 0, y: 0 },
            data: { title: "Shared task" },
          },
        ],
      },
    });

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      onConnect={onConnect}
      nodeTypes={{ task: TaskNodeView }}
    />
  );
}
```

## Cursors [#cursors]

The package also exports a `Cursors` component for collaborative pointers on the
flow canvas. Render it **inside** [`ReactFlow`](https://reactflow.dev/) (as a
child, like other flow overlays) so it can read the viewport. Import
`@liveblocks/react-flow/styles.css` when using it.

```tsx highlight="1,16"
import { Cursors, useLiveblocksFlow } from "@liveblocks/react-flow/suspense";
import { ReactFlow } from "@xyflow/react";

function Flow() {
  const { nodes, edges, onNodesChange, onEdgesChange, onConnect } =
    useLiveblocksFlow();

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      onConnect={onConnect}
    >
      <Cursors />
    </ReactFlow>
  );
}
```
